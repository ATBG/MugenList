2. Complete Season Interface (TypeScript + JSON Schema Ready)
interface Season {
  // ── CORE IDENTIFIERS (API-sourced, immutable after creation) ──
  mal_id: number;                    // Season-specific MyAnimeList ID (unique per season entry)
  season_number: number;             // 1-based sequential number within franchise (S1=1, S2=2, etc.)
  season_name: string;               // Display name: "Season 2", "Cour 1", "Summer 2025", "OVA", "Movie", etc.
  season_year: number;               // Year the season originally aired or is scheduled (e.g. 2025)

  // ── AIRING STATUS (real-time critical) ──
  is_airing: boolean;                // true if season is currently broadcasting new episodes
  aired_episodes: number;            // Episodes that have actually aired so far (updated live)
  total_episodes: number | null;     // Total planned episodes (null = unknown / ongoing)

  // ── NEXT EPISODE COUNTDOWN (per-season only – never on root) ──
  next_episode_airing_at: string | null;   // ISO 8601 UTC timestamp or null if not airing
  next_episode_number: number | null;      // The episode number that will air next (e.g. 14)

  // ── USER PROGRESS & LIBRARY STATE (per-season – critical for Continue Watching & recommendations) ──
  user_status: 'watching' | 'completed' | 'planned' | 'dropped' | 'on_hold' | null;
  watched_episodes: number;          // User-tracked episodes watched for THIS season only
  last_watched_at: string | null;    // ISO timestamp of last user interaction with this season
  user_rating: number | null;        // 1-10 or null (optional per-season rating)
  user_notes: string | null;         // Optional free-text notes for this season

  // ── VISUAL & METADATA (for cards, rails, hero) ──
  poster_url: string;                // Full 2:3 poster image URL (MAL or AniList)
  banner_url: string | null;         // Optional wide banner for hero/featured
  synopsis_short: string;            // 1-2 sentence synopsis specific to this season
  genres: string[];                  // Array of genre strings (inherited + season-specific if available)
  score: number | null;              // MAL/AniList score for this season (0-10)

  // ── INTERNAL APP FIELDS (for caching, sync, UI state) ──
  last_synced_at: string;            // ISO timestamp of last successful MAL/AniList refresh for this season
  sync_status: 'synced' | 'pending' | 'failed' | 'airing-live';
  relation_type: 'sequel' | 'prequel' | 'side_story' | 'parent' | null; // From franchise resolution
  absolute_order: number;            // Global ordering index within franchise (used for "latest season first" sorting)
}
